if __name__ == '__main__':
    move_forward = MoveForward()
    while True:
        input_str = input("Input Goal Type, goal type (F/R/L),(e.g. 0.5,F,5): ")
        input_list = input_str.split(',')
        if input_list[1] == 'F':
            goal_type = "forward"
            goal_amount = float(input_list[0])
            num_moves = int(input_list[2])
            for i in range(num_moves):
                current_pose = move_forward.get_current_pose()
                distance_to_move = goal_amount
                if current_pose is not None:
                    distance_to_move += current_pose.position.x
                move_forward.move_forward(distance_to_move)
        elif input_list[1] == 'R':
            goal_type = "right rotation"
            goal_amount = 90
            num_moves = int(input_list[2])
            for i in range(num_moves):
                move_forward.move_right_rotation(goal_amount)
        elif input_list[1] == 'L':
            goal_type = "left rotation"
            goal_amount = 90
            num_moves = int(input_list[2])
            for i in range(num_moves):
                move_forward.move_left_rotation(goal_amount)
        elif input_list[0] == 'END':
            break
        else:
            print("Invalid input.")
            continue
        print("Reached {} goal of {} {} for {} times.".format(goal_type, goal_amount, "meter" if goal_type == "forward" else "degrees", num_moves))
